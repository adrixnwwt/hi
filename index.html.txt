<script>
async function sendMessage() {
  const message = document.getElementById("message").value;

  await fetch("https://DEINE-VERCEL-URL.vercel.app/api/send", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ message })
  });

  alert("Message sent!");
}
</script>
